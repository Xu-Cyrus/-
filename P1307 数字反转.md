
### P1307 数字反转

```cpp
#include <iostream>
#include <cstdio>
using namespace std;
int main()
{
    bool b=false;
    int a;
    scanf("%d",&a);
    if(a<0)
    {
        printf("-");
        a=a*-1;
    }
    if(a==0) printf("0");
    while(a>0)
    {
        if(a%10!=0)
        {
            b=true;
        }
        if(b==false&&a%10==0)
        {
            a=a/10;
            continue;
        }
        printf("%d",a%10);
        a=a/10;
    }
    return 0;
}
```
