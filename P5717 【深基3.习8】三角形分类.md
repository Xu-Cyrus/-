
### P5717 【深基3.习8】三角形分类

```cpp
#include <iostream>
#include <cstdio>
#include <algorithm>
using namespace std;
int v[2];
int main()
{
    scanf("%d%d%d",&v[0],&v[1],&v[2]);
    sort(v,v+3);
    int a,b,c;
    a=v[0];
    b=v[1];
    c=v[2];
    int x,y,z;
    x=a*a;
    y=b*b;
    z=c*c;
    if(a+b<c||a+c<b||b+c<a)
    {
        printf("Not triangle");
        return 0;
    }
    if(x+y==z||x+z==y||y+z==x)
    {
        printf("Right triangle\n");
    }
    if(x+y>z)
    {
        printf("Acute triangle\n");
    }
    if(x+y<z)
    {
        printf("Obtuse triangle\n");
    }
    if(a==b||a==c||b==c)
    {
        printf("Isosceles triangle\n");
    }
    if(a==b&&a==c&&b==c)
    {
        printf("Equilateral triangle\n");
    }
    return 0;
}

```
