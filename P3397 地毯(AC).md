
### P3397 地毯(AC)

```cpp
#include <iostream>
#include <cstdio>
#include <cstring>
using namespace std;
int main()
{
    unsigned int n,m,i,x1,y1,x2,y2,o,p,j;
    scanf("%d%d",&n,&m);
    int a[n+1][n+1];
    memset(a,0,sizeof(a));
    for(i=1;i<=m;i++)
    {
        scanf("%d%d%d%d",&x1,&y1,&x2,&y2);
        for(o=x1;o<=x2;o++)
            for(p=y1;p<=y2;p++)
            {
                a[o][p] ++;
            }
    }
    for(i=1;i<=n;i++)
    {
        for(j=1;j<=n;j++)
        {
            printf("%d ",a[i][j]);
        }
        printf("\n");
    }
    return 0;
}
```
